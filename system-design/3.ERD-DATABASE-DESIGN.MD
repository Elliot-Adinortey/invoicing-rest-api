        int id PK
        string name
        string email
        string password
        datetime created_at
        datetime updated_at
    }

    CUSTOMERS {
        int id PK
        int user_id FK
        string customer_name
        string email
        string phone
        string address
        datetime created_at
        datetime updated_at
    }

    PRODUCTS {
        int id PK
        int user_id FK
        string product_name
        decimal unit_price
        int stock_quantity
        datetime created_at
        datetime updated_at
    }

    INVOICES {
        int id PK
        int user_id FK
        int customer_id FK
        string invoice_number
        date issue_date
        date due_date
        string status
        decimal subtotal
        decimal tax
        decimal total_amount
        datetime created_at
        datetime updated_at
    }

    INVOICE_ITEMS {
        int id PK
        int invoice_id FK
        int product_id FK
        int quantity
        decimal unit_price
        decimal subtotal
        string description
    }

    STOCK_MOVEMENTS {
        int id PK
        int product_id FK
        int invoice_id FK
        string movement_type

    }
